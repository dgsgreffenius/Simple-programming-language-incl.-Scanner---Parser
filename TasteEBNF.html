EBNF Grammar of Taste

Taste =      "program" ident "{" {ConstDecl} {VarDecl} {ProcDecl} "}".
VarDecl =    Type ident [ '[' number ']' ] {"," ident [ '[' number ']' ]} ";".
ConstDecl =  "const" ident ":=" Expr {"," ConstDecl} ";".
ProcDecl =   "void" ident "(" ")" "{" {ConstDecl | VarDecl | Stat} "}".
Type =       "int" | "bool" | "const".

Stat =       ident ( ( '[' number ']' | )
					":=" Expr ["?" Expr ":" Expr ]
					|"(" ")"
					) ";"
           | "if" "(" Expr ")" Stat ["else" Stat]
           | "while "(" Expr ")" Stat
		   | "do" Stat "while" "(" Expr ");"
           | "read" ident ";"
           | "write" Expr {", " Expr} ";"
           | "{" {Stat | VarDecl} "}"
		   | "for" ident ":=" (number | ident) "to" (number | ident) "do" Stat.

Expr =       SimExpr [RelOp SimExpr].
SimExpr =    Term {AddOp Term}.
Term =       Factor {MulOp Factor}.
Factor =     ( "(" Expr ")" | ident( '[' number ']' |) | "true" | "false" | number | "-" Factor).

AddOp =      "+" | "-".
MulOp =      "*" | "/".
RelOp =      "==" | "<" | ">" | "!=" | "<=" | ">=".
